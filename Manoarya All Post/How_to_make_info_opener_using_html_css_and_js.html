Add HTML Code

 
<!DOCTYPE html>
<head>

	content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=0">

	<p id="info1">

	</p>

		<!--Add Your Own text -->
	</p>

		<!--Add Your Own text -->
	</p>
</body>

Add Javascript Code

       
//get info
    const info1 = document.getElementById('info1');
    const info2 = document.getElementById('info2');
    const info3 = document.getElementById('info3');
//get span + -
    const icons1 = document.getElementById('icons1');
    const icons2 = document.getElementById('icons2');
    const icons3 = document.getElementById('icons3');
//onclick function
    function info1btn() {
      info1.classList.toggle('open');
      icons1.classList.toggle('icons');
    }
    function info2btn() {
      info2.classList.toggle('open');
      icons2.classList.toggle('icons');
    }
    function info3btn() {
      info3.classList.toggle('open');
      icons3.classList.toggle('icons');
    }
