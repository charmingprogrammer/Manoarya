Add Javascript Code

          
    //Access All Necessary  items
    const big_Image = document.getElementById('big_Image');
    const pic1 = document.getElementById('pic1');
    const pic2 = document.getElementById('pic2');
    const pic3 = document.getElementById('pic3');
    const pic4 = document.getElementById('pic4');
    const pic5 = document.getElementById('pic5');


    // All functions
    function p1() {
      big_Image.innerHTML = pic1.innerHTML;
    }
    function p2() {
      big_Image.innerHTML = pic2.innerHTML;
    }
    function p3() {
      big_Image.innerHTML = pic3.innerHTML;
    }
    function p4() {
      big_Image.innerHTML = pic4.innerHTML;
    }
    function p5() {
      big_Image.innerHTML = pic5.innerHTML;
    }

    //Onclick event on all small image.
    pic1.addEventListener('click', p1);
    pic2.addEventListener('click', p2);
    pic3.addEventListener('click', p3);
    pic4.addEventListener('click', p4);
    pic5.addEventListener('click', p5);
